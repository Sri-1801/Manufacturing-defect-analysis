# Convert 'Date' column to datetime format (if not already)
df['Date'] = pd.to_datetime(df['Date'])

# Create a new column with month-year
df['Month'] = df['Date'].dt.to_period('M')

# Count of defects per month
monthly_defects = df['Month'].value_counts().sort_index()

print("📈 Defects per Month:")
print(monthly_defects)

# Plotting
monthly_defects.plot(kind='line', marker='o', title='Monthly Defect Trend', figsize=(10, 5))
plt.xlabel('Month')
plt.ylabel('Number of Defects')
plt.grid(True)
plt.tight_layout()
plt.show()
